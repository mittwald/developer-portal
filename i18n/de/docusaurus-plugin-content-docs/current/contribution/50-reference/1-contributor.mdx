import SchemaWithExample from "../../../../../../src/components/openapi/SchemaWithExample";

Contributoren werden derzeit über statische YAML-Dokumente definiert.
Die benötigten und erlaubten Attribute sind in folgendem Schema beschrieben.

<SchemaWithExample
  withRawJSONSchema
  format="yaml"
  schema={{
    type: "object",
    required: ["id", "name", "customerId", "state", "supportInformation", "contributorNumber", "imprint"],
    properties: {
      id: {
        type: "string",
        format: "uuid",
        description: "ID of the contributor. It has to be globally unique.",
      },
      name: {
        type: "string",
        description: "Name of the contributor.",
        example: "mittwald",
      },
      customerId: {
        type: "string",
        format: "uuid",
        description:
          "ID of the organization in the mStudio, the contributor belongs to. The mapping of contributors to organizations is 1:1.",
      },
      supportInformation: {
        type: "object",
        properties: {
          email: {
            type: "string",
            example: "support@mittwald.de"
          },
          phone: {
            type: "string",
            example: "+49 (0) 5772/293-600"
          }
        }
      },
      homepage : {
        type: "string",
        description: "Homepage of the Contributor",
        example: "https://mittwald.de/"
      },
      logoRefId: {
        type: "string",
        example: "79d92944-fe9b-48bd-90c4-24048f3269a3"
      },
      state: {
        type: "object",
        description: "State of the Contributor.",
        example: {
          verified: true
        },
        properties: {
          disabled: {
            type: "boolean",
          },
        },
      },
      descriptions: {
        type: "object",
        properties: {
          de: {
            type: "string",
            example: "Wir leben Web. Wir atmen Code. Und: Wir lieben Agenturen & Freelancer. Mit maximal performantem und flexiblem Managed Cloud Hosting bieten wir Web Professionals die Plattform, auf der sie mit ihren Webprojekten durchstarten. Clevere Workflows und smarte Features, wie die Analyse der Time to First Byte, machen deine Arbeit effizienter. So macht Hosting Spaß!"
          },
          en: {
            type: "string",
            example: "We live the web. We breathe code. And: We love agencies & freelancers. With maximum performance and flexible managed cloud hosting, we offer web professionals the platform on which they can take off with their web projects. Clever workflows and smart features, such as time-to-first byte analysis, make your work more efficient. This makes hosting fun!"
          }
        }
      },
      contributorNumber: {
        type: "string",
        example: "00000001"
      },
      imprint: {
        oneOf: [
          {
            type: "object",
            properties: {
              text: {
                type: "string",
                example: "Mittwald CM Service GmbH & Co. KG\nKönigsberger Straße 4-6\n32339 Espelkamp\nTelefon: +49-5772-293-10\nTelefax: +49-5772-293-33\nHRA: 6640, AG Bad Oeynhause\nUSt. ID-Nr: DE81477321\nKomplementärin: Robert Meyer Verwaltungs GmbH, AG Bad Oeynhausen HRB 13260\nvertreten durch die Geschäftsführer: Robert Meyer, Florian Jürgens"
              }
            },
            required: [ "text" ]
          },
          {
            type: "object",
            properties: {
              url: {
                type: "string"
              }
            },
            required: [ "url" ]
          }
        ]
      }
    },
  }}
/>
