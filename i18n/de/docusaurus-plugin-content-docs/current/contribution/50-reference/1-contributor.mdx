---
title: Contributoren
---

import SchemaWithExample from "../../../../../../src/components/openapi/SchemaWithExample";

Contributoren werden derzeit über statische YAML-Dokumente definiert.
Die benötigten und erlaubten Attribute sind in folgendem Schema beschrieben.

<SchemaWithExample
  withRawJSONSchema
  format="yaml"
  schema={{
    type: "object",
    required: ["id", "name", "customerId", "state", "email"],
    properties: {
      id: {
        type: "string",
        format: "uuid",
        description: "ID des Contributoren. Muss global eindeutig sein.",
      },
      name: {
        type: "string",
        description: "Name des Contributoren.",
        example: "mittwald",
      },
      description: {
        type: "string",
        description: "Kurzbeschreibung des Contributoren.",
        example:
          "Deutscher Web-Hoster und Contributor von Extensions für das mStudio.",
      },
      customerId: {
        type: "string",
        format: "uuid",
        description:
          "ID der Organisation im mStudio, zu der der Contributor gehört. Ein Contributor hat ein 1:1 Verhältnis zu einer Organisation",
      },
      state: {
        type: "object",
        description: "Status des Contributoren.",
        example: {
          disabled: false,
        },
        properties: {
          disabled: {
            type: "boolean",
          },
        },
      },
      homepage: {
        type: "string",
        description: "Homepage des Contributors",
        example: "https://mittwald.de/",
      },
      email: {
        type: "string",
        format: "email",
        description:
          "E-Mail-Adresse des Contributoren, die für Support Zwecke genutzt werden kann.",
      },
      phone: {
        type: "string",
        format: "phone",
        example: "+49123456789",
        description:
          "Telefonnummer des Contributoren, die für Support Zwecke genutzt werden kann.",
      },
      imprint: {
        oneOf: [
          {
            type: "object",
            properties: {
              text: {
                type: "string",
                example:
                  "Innovate Solutions GmbH & Co. KG\nTech Park 8\n10115 Berlin\nPhone: +49-30-1234-567\nFax: +49-30-1234-568\nHRA: 7789, AG Berlin (Charlottenburg)\nUSt. ID-Nr: DE987654321\nKomplementärin: Alpha Management GmbH, AG Berlin (Charlottenburg) HRB 24680\nVertreten durch die Geschäftsführer: Dr. Lena Müller, David Schmidt\n",
              },
            },
            required: ["text"],
          },
          {
            type: "object",
            properties: {
              url: {
                type: "string",
              },
            },
            required: ["url"],
          },
        ],
      },
    },
  }}
/>
